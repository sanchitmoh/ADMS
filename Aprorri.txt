transactions <- list(
  c("Milk","Bread","Butter"),
  c("Bread","Butter"),
  c("Milk","Bread"),
  c("Milk","Butter"),
  c("Bread","Butter","Jam")
)
install.packages("arules")   # only once
library(arules)

trans <- as(transactions, "transactions")
trans

rules <- apriori(
  trans,
  parameter = list(
    supp = 0.4,
    conf = 0.6
  )
)
inspect(rules)

//sortedbyconidenec
rules_conf <- sort(rules, by = "confidence", decreasing = TRUE)
inspect(rules_conf)
//sorted bylift
rules_lift <- sort(rules, by = "lift", decreasing = TRUE)
inspect(rules_lift)
//Filtering Strong Rules
strong_rules <- subset(rules, lift > 1)
inspect(strong_rules)
