//regression
data <- data.frame(
  hours = c(2,4,6,8,10),
  marks = c(40,50,65,75,85)
)
data

plot(data$hours, data$marks,
     main = "Hours Studied vs Marks",
     xlab = "Hours Studied",
     ylab = "Marks",
     pch = 16)
model <- lm(marks ~ hours, data = data)
model
summary(model)
coef(model)
abline(model, col = "blue", lwd = 2)
predict(model, data.frame(hours = 7))
residuals(model)
